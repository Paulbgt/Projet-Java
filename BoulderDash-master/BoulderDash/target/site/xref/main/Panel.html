62      	/**
63  	 * paintComponent this function allows many functionality
64  	 * convert the string table into picture table 
65  	 * monter's Movement, this is random
66  	 * kill monster
67  	 * push stone in any direciton
68  	 * detection of stone, diamond and wall
69  	 * diamond and stone's gravity
70  	 * display the different element, rockford, score  ans life
71  	 * 
72  	 * 
73  	 * 
74  	 */
75  	public void paintComponent(Graphics g){
76  		try{	
77  			// Lists containing the position (x and y) of the diamonds, rocks and monsters
78  		    ArrayList<Integer> Listx_stone = new ArrayList<Integer>();
79  		    ArrayList<Integer> Listy_stone = new ArrayList<Integer>();
80  		    ArrayList<Integer> Listx_diams = new ArrayList<Integer>();
81  		    ArrayList<Integer> Listy_diams = new ArrayList<Integer>();
82  		    ArrayList<Integer> Listx_monster = new ArrayList<Integer>();
83  		    ArrayList<Integer> Listy_monster = new ArrayList<Integer>();
84  		    ArrayList<Integer> Listx_monster2 = new ArrayList<Integer>();
85  		    ArrayList<Integer> Listy_monster2 = new ArrayList<Integer>();
86  			// Retrieving the String table from the controller
87  			if(first==0){tbl=controller.start(Nmap);first++;}
88  			// Converting the string array to an image table
89  			for(int y=0;y<30;y++){
90  				for(int x=0; x<40; x++){
91  						tbl[x][y]=tbl[x][y].intern();
92  						if(tbl[x][y]=="#"){tbl_image[x][y]=img_wall;}
93  						else if(tbl[x][y]=="0"){tbl_image[x][y]=img_durt;}
94  						else if(tbl[x][y]=="S"){tbl_image[x][y]=img_stone;Listx_stone.add(x);Listy_stone.add(y);}
95  						else if(tbl[x][y]=="B"){tbl_image[x][y]=img_background;}
96  						else if(tbl[x][y]=="D"){tbl_image[x][y]=img_diamond;Listx_diams.add(x);Listy_diams.add(y);}
97  						else if(tbl[x][y]=="M"){tbl_image[x][y]=img_monster;Listx_monster.add(x);Listy_monster.add(y);}
98  						else if(tbl[x][y]=="M2"){tbl_image[x][y]=img_monster2;Listx_monster2.add(x);Listy_monster2.add(y);}
99  						else{System.out.println("Unknown character !");;}
100 					}
101 				
102 				}
103 			
104 			
105 			// Monster Movements
106 			int list=0;
107 			int size_monster=Listx_monster.size();
108 			int move=0;
109 			
110 				// Generating a random movement
111 			while(list<size_monster){
112 			move = (int) (Math.random() * 5 );
113 			
114 				// Kill the Monster 1
115 			if(move==3){if(tbl[Listx_monster.get(list)][Listy_monster.get(list)-1]=="S" || tbl[Listx_monster.get(list)][Listy_monster.get(list)-1]=="D"){tbl[Listx_monster.get(list)][Listy_monster.get(list)]="D";}}
116 				// Monster 1 goes down
117 			if(move==1){if(tbl[Listx_monster.get(list)][Listy_monster.get(list)+1]=="B"){
118 				tbl[Listx_monster.get(list)][Listy_monster.get(list)+1]="M";
119 				tbl[Listx_monster.get(list)][Listy_monster.get(list)]="B";
120 				Listy_monster.set(list, Listy_monster.get(list)+1);}}
121 				// Monster 1 goes right
122 			if(move==2){if(tbl[Listx_monster.get(list)+1][Listy_monster.get(list)]=="B"){
123 				tbl[Listx_monster.get(list)+1][Listy_monster.get(list)]="M";
124 				tbl[Listx_monster.get(list)][Listy_monster.get(list)]="B";
125 				Listx_monster.set(list, Listx_monster.get(list)+1);}}
126 				// Monster 1 goes up
127 			if(move==3){if(tbl[Listx_monster.get(list)][Listy_monster.get(list)-1]=="B"){
128 				tbl[Listx_monster.get(list)][Listy_monster.get(list)-1]="M";
129 				tbl[Listx_monster.get(list)][Listy_monster.get(list)]="B";
130 				Listy_monster.set(list, Listy_monster.get(list)-1);}}
131 				// Monster 1 goes left
132 			if(move==4){if(tbl[Listx_monster.get(list)-1][Listy_monster.get(list)]=="B"){
133 				tbl[Listx_monster.get(list)-1][Listy_monster.get(list)]="M";
134 				tbl[Listx_monster.get(list)][Listy_monster.get(list)]="B";
135 				Listx_monster.set(list, Listx_monster.get(list)-1);}}
136 			list++;
137 			}
138 			
139 			// Movement of the Monster 2 (he deposits a stone before moving randomly in the map when the player goes to the right)
140 			if(state==4){
141 			int size_monster2=Listx_monster2.size();
142 			int moveX=0;
143 			int moveY=0;
144 			list=0;
145 			while(list<size_monster2){
146 			moveX = (int) (Math.random() * 39 );
147 			moveY = (int) (Math.random() * 29 );
148 			if(moveX!=0 && moveY!=0 ){
149 				tbl[Listx_monster2.get(list)][Listy_monster2.get(list)]="S";
150 				tbl[moveX][moveY]="M2";Listx_monster2.set(list, moveX);Listy_monster2.set(list, moveY);}list++;}}
151 
152 			
153 			// Push the stones
154 				// to the right
155 			list=0;
156 			if(state==4){if(tbl[PositionX][PositionY]=="S" && tbl[PositionX+1][PositionY]=="B"){
157 				tbl[PositionX][PositionY]="B";
158 				tbl[PositionX+1][PositionY]="S";
159 				while(PositionX!=Listx_stone.get(list)){list++;}
160 				Listx_stone.set(list,Listx_stone.get(list)+1);
161 				}
162 			}
163 				// to the left
164 			list=0;
165 			if(state==3){if(tbl[PositionX][PositionY]=="S" && tbl[PositionX-1][PositionY]=="B"){
166 				tbl[PositionX][PositionY]="B";
167 				tbl[PositionX-1][PositionY]="S";
168 				while(PositionX!=Listx_stone.get(list)){list++;}
169 				Listx_stone.set(list,Listx_stone.get(list)-1);
170 				}
171 			}
172 				
173 
174 			// Stop the collision against rocks and walls
175 			if(state==1){if(tbl[PositionX][PositionY]=="S" || tbl[PositionX][PositionY]=="#"){PositionY++;}}
176 			if(state==2){if(tbl[PositionX][PositionY]=="S" || tbl[PositionX][PositionY]=="#"){PositionY--;}}
177 			if(state==3){if(tbl[PositionX][PositionY]=="S" || tbl[PositionX][PositionY]=="#"){PositionX++;}}
178 			if(state==4){if(tbl[PositionX][PositionY]=="S" || tbl[PositionX][PositionY]=="#"){PositionX--;}}
179 			
180 
181 			// Falling stones in cascade
182 			size_stone=Listx_stone.size();
183 			size_diams=Listx_diams.size();
184 			
185 			while(list<size_stone){
186 			if(tbl[Listx_stone.get(list)][Listy_stone.get(list)+1]=="S"){
187 				// Look at the bottom right
188 				if(tbl[Listx_stone.get(list)+1][Listy_stone.get(list)+1]=="B"){
189 					tbl[Listx_stone.get(list)][Listy_stone.get(list)]="B";
190 					tbl[Listx_stone.get(list)+1][Listy_stone.get(list)+1]="S";
191 					Listx_stone.set(list,Listx_stone.get(list)+1);
192 					Listy_stone.set(list,Listy_stone.get(list)+1);
193 					}
194 				// Look at the bottom left
195 				if(tbl[Listx_stone.get(list)-1][Listy_stone.get(list)+1]=="B"){
196 					tbl[Listx_stone.get(list)][Listy_stone.get(list)]="B";
197 					tbl[Listx_stone.get(list)-1][Listy_stone.get(list)+1]="S";
198 					Listx_stone.set(list,Listx_stone.get(list)-1);
199 					Listy_stone.set(list,Listy_stone.get(list)+1);
200 					}
201 			}
202 			list++;
203 			}
204 
205 			
206 			// Falling stones (gravity)
207 			list=0;
208 			while(list<size_stone){
209 			if(tbl[Listx_stone.get(list)][Listy_stone.get(list)+1]=="B"){
210 				tbl[Listx_stone.get(list)][Listy_stone.get(list)+1]="S";
211 				tbl[Listx_stone.get(list)][Listy_stone.get(list)]="B";
212 				Listy_stone.set(list,Listy_stone.get(list)+1);}
213 			list++;
214 			}
215 			
216 			// Detection of diamond / collection
217 			if(state==1 || state==2 || state==3 || state==4){if(tbl[PositionX][PositionY]=="D"){state=0;Cmpt_Diams++;}}
218 			// Perte de vie
219 			if(state==1 || state==2 || state==3 || state==4){if(tbl[PositionX][PositionY]=="M" || tbl[PositionX][PositionY]=="S"){Life--;state=0;}}
220 
221 
222 			// Falling diamond
223 			for(int list1=0;list1<size_diams;list1++){
224 			if(tbl[Listx_diams.get(list1)][Listy_diams.get(list1)+1]=="B"){
225 				tbl[Listx_diams.get(list1)][Listy_diams.get(list1)]="B";
226 				tbl[Listx_diams.get(list1)][Listy_diams.get(list1)+1]="D";
227 				Listy_diams.set(list1,Listy_diams.get(list1)+1);}
228 			}
229 
230 
231 			// Display of the picture table
232 			
233 				for(int y=0;y<30;y++){
234 					for(int x=0; x<40; x++){
235 						g.drawImage(tbl_image[x][y].getImage(), x*32, y*32,32,32,this);
236 						g.drawImage(img_rockford.getImage(), PositionX*32, PositionY*32,32,32,this);
237 					}
238 				}
239 				
240 			// Finished Games
241 				
242 				size_diams=Listx_diams.size();
243 				
244 				
245 				
246 				if (WinGame == 1){
247 
248 					Thread.sleep(3000);
249 					String nb="0";
250 					Window windo = new Window(nb);
251 					windo.exit();
252 			}
253 						// YOU WIN !
254 				if(size_diams==0){
255 					for(int y=0;y<30;y++){
256 						for(int x=0; x<40; x++){
257 							g.drawImage(img_durt.getImage(), x*32, y*32,32,32,this);}}
258 					
259 						WinGame = 1;	
260 							g.setColor(Color.BLUE);
261 							g.setFont(new Font("arial", Font.BOLD,100));
262 							g.drawString("YOU WIN !", 350,450);
263 					}
264 					// YOU LOOSE !
265 				if(Life==0){
266 					for(int y=0;y<30;y++){
267 						for(int x=0; x<40; x++){
268 							g.drawImage(img_durt.getImage(), x*32, y*32,32,32,this);}}
269 					
270 						WinGame = 1;	
271 							g.setColor(Color.BLUE);
272 							g.setFont(new Font("arial", Font.BOLD,100));
273 							g.drawString("YOU LOOSE !", 350,450);
274 					}
275 
276 
277 		
278 			// Hero Display
279 				
280 			g.drawImage(img_rockford.getImage(), PositionX*32, PositionY*32,32,32,this);
281 			
282 			// Score display
283 			g.setColor(Color.RED);
284 			g.setFont(new Font("arial", Font.BOLD,28));
285 			g.drawString("Diams : "+Cmpt_Diams, 9,30);
286 			// Life display
287 			g.setColor(Color.RED);
288 			g.setFont(new Font("arial", Font.BOLD,28));
289 			g.drawString("Life : "+Life, 200,30);
290 			
291 			state=0;
292 			
293 
294 		}catch (SQLException e) {e.printStackTrace();} catch (InterruptedException e) {
295 			// TODO Auto-generated catch block
296 			e.printStackTrace();
297 		}

300 	// Different player movements
301 	
302 	public void Up(){ 	
303 			tbl[PositionX][PositionY]="B";
304 			PositionY--;
305 			state=1;
306 	}
307 	public void down(){
308 		tbl[PositionX][PositionY]="B";
309 		PositionY++;
310 		state=2;
311 	}
312 	public void left(){
313 		tbl[PositionX][PositionY]="B";
314 		PositionX--;
315 		state=3;
316 	}
317 	public void right(){
318 		tbl[PositionX][PositionY]="B";
319 		PositionX++;
320 		state=4;
321 	}
